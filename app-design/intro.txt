        <div className="leaf"></div>
        <div className="leaf"></div>
        <div className="leaf"></div>
      </div>
      <div className="title-container">
        <div className="ammu-text">
          {Array.from("Ammu's").map((char, index) => (
            <motion.span
              key={index}
              initial={{ opacity: 0, y: 50, scale: 0.8 }}
              animate={{ opacity: 1, y: 0, scale: 1 }}
              transition={{ delay: index * 0.15, duration: 0.6, ease: "easeOut" }}
              className="letter ammu"
            >
              {char}
            </motion.span>
          ))}
        </div>
        <div className="diary-text">
          {Array.from("Diary").map((char, index) => (
            <motion.span
              key={index}
              initial={{ opacity: 0, y: 50, scale: 0.8 }}
              animate={{ opacity: 1, y: 0, scale: 1 }}
              transition={{ delay: (index + 7) * 0.15, duration: 0.6, ease: "easeOut" }}
              className="letter diary"
            >
              {char}
            </motion.span>
          ))}
        </div>
      </div>
      <div className="nature-bottom">
        <div className="curved-nature"></div>
        <div className="curved-line curved-line-1"></div>
        <div className="curved-line curved-line-2"></div>
        <div className="curved-line curved-line-3"></div>
        
        {/* Original flower stems */}
        <div className="flower-stem flower-stem-1"></div>
        <div className="flower-stem flower-stem-2"></div>
        <div className="flower-stem flower-stem-3"></div>
        
        {/* New flower groups */}
        {[1, 2, 3, 4].map((groupNum) => (
          <div key={`group-${groupNum}`} className={`flower-group flower-group-${groupNum}`}>
            <div className="stem">
              <div className="bloom-1 flower-bloom-small"></div>
              <div className="bloom-2 flower-bloom-small"></div>
              <div className="bloom-3 flower-bloom-small"></div>
              {[...Array(4)].map((_, i) => (
                <div
                  key={`leaf-${i}`}
                  className={`leaf-small ${i % 2 ? 'right' : ''}`}
                  style={{
                    top: `${20 + i * 15}px`,
                    left: i % 2 ? '2px' : '-15px',
                    animationDelay: `${0.5 + i * 0.2}s`
                  }}
                ></div>
              ))}
            </div>
          </div>
        ))}

        {/* Enhanced floating petals */}
        {[...Array(20)].map((_, i) => (
          <div
            key={`petal-fancy-${i}`}
            className={`petal-fancy ${
              i % 3 === 0 ? 'pink-light' : i % 3 === 1 ? 'purple' : ''

              left: `${Math.random() * 80 + 10}%`,
              bottom: `${Math.random() * 100 + 50}px`,
              '--float-x': `${Math.random() * 200 - 100}px`,
              '--float-y': `${Math.random() * -200}px`,
              '--rotate': `${Math.random() * 720 - 360}deg`,
              animationDelay: `${Math.random() * 8}s`
            }}
          ></div>
        ))}
      </div>
    </div>
  );
};

export default Intro;
